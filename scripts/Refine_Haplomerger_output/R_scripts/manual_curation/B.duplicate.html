The plots below show the MUMmer alignement between the haploid genome assembly ZC02 and the haplomerger output ZC04 to identify remaining redundant sequences. It also includes the results of FindBreakingPoint() algorithm to estimate the regions to be trimmed. The dot plot have the following structure:

  • dots represent short high identity alignments (~1500 bp)
  • x axis shows the starting position of these alignments in the reference scaffold (ZC02)
  • y axis shows the starting position of these alignments in the query scaffold (ZC04)
  • Color correspond to unique scaffolds, each color correspond an alignment to a different reference scaffold
  • maroon lines correspond to the limits identified by FindBreakingPoint() of the areas to be trimmed
# Load data ...............................................

setwd( '~/Documents/Projects/ZC07/R' )
mummer <- read.table('B1.coords.txt')
colnames(mummer) <- c('RS','RE','QS','QE','L1','L2','Percent'
                      ,'lenR','lenQ','frmR','frmQ','reference'
                      ,'query')

B.zc04_sizes <- read.table('B.ZC04.sizes',row.names = 1)
B.zc04       <- B.zc04_sizes$V2

# Read scoring matrix ..............................................

scores <- read.table('B2.scores.txt', row.names = 1)
colnames(scores) <- c('int','cont','len')

# Read Breaking points .............................................

coords <- read.table('B3.breakpoints.txt', row.names = 1)
colnames(coords) <- c('start','end')

# Cathegorize scaffolds based on contintuity and integrity ................

mosaic1 <- scores[scores$len > 5 & scores$len < 10,]
mosaic2 <- scores[scores$len >= 10 & scores$int < 0.95, ]
mosaic  <- rbind(mosaic1, mosaic2)

linear1 <- scores[scores$len <= 5,]
linear2 <- scores[scores$len >= 10 & scores$int >= 0.95 & scores$cont >= 0.95,]
linear <- rbind(linear1, linear2)

duplicated <- scores[scores$len >= 10 & scores$int >=0.95 & scores$cont < 0.95,]
# Create lists of alignments ...................................................

scaff_m <- as.vector(row.names(mosaic))
scaff_l <- as.vector(row.names(linear))
scaff_d <- as.vector(row.names(duplicated))
scaff_a <- B.zc04[!B.zc04 %in% row.names(scores)]

# Write lists to text files ....................................................

write(scaff_m, 'Automated_cathegorization/B.names.mosaic')
write(scaff_l, 'Automated_cathegorization/B.names.linear')
write(scaff_d, 'Automated_cathegorization/B.names.duplicate')
write(scaff_a, 'Automated_cathegorization/B.names.unaligned')

man_dup <- as.vector(read.table('../Manual_curation/B.names.duplicate')$V1)
man_linear <- as.vector(unique(mummer$query)[! unique(mummer$query) %in% man_dup])
# Plot .........................................................................

for( scaff in man_dup) { PlotScaffoldAlignment( mummer, scaff )
                        abline(coords[scaff,'start'],0, col = 'brown4', lwd = 2)
                        abline(coords[scaff,'end'],0, col = 'brown4', lwd = 2)
                        }
